### 12. goroutine
goroutine 是由 Go 运行时环境管理的轻量级线程。

```go
package main

import (
  "fmt"
  "time"
)

func say(s string) {
  for i := 0; i < 5; i++ {
    time.Sleep(100 * time.Millisecond)
    fmt.Println(s)
  }
}

func main() {
  go say("world") //开启一个新的goroutine执行
  say("hello")
}
```

### 13. channel

```go
package main

import "fmt"

func sum(a []int, c chan int) {
  sum := 0
  for _, v := range a {
    sum += v
  }
  c <- sum // 将和送入 c, 箭头方向为数据流方向
}

func main() {
  a := []int{7, 2, 8, -9, 4, 0}

  c := make(chan int) // 使用前需要使用make创建
  go sum(a[:len(a)/2], c)
  go sum(a[len(a)/2:], c)
  x, y := <-c, <-c // 从 c 中获取

  fmt.Println(x, y, x+y)
}
```
